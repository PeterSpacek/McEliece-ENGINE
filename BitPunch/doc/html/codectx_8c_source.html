126 int BPU_codeInitMathCtx(BPU_T_Math_Ctx **ctx, const uint16_t m, const uint16_t t, const BPU_T_GF2_16x mod) {
127  int rc = 0;
128 
129  if (mod == (BPU_T_GF2_16x) -1) {
130  *ctx = (BPU_T_Math_Ctx *) calloc(1, sizeof(BPU_T_Math_Ctx));
131  if (!*ctx) {
132  BPU_printError("Can not malloc BPU_T_Math_Ctx");
133 
134  return -1;
135  }
136  (*ctx)->mod_deg = m;
137  }
138  else if (mod != 0) {
139  rc = BPU_mathInitCtx(ctx, (BPU_T_GF2_16x)2, mod);
140  }
141  else if (m == 5 && t == 5) {
142  rc = BPU_mathInitCtx(ctx, (BPU_T_GF2_16x)2, (BPU_T_GF2_16x) BPU_GF2_POLY_DEG_5);
143  }
144  else if (m == 6 && t == 6) {
145  rc = BPU_mathInitCtx(ctx, (BPU_T_GF2_16x)2, (BPU_T_GF2_16x) BPU_GF2_POLY_DEG_6);
146  }
147  else if (m == 6 && t == 7) {
148  rc = BPU_mathInitCtx(ctx, (BPU_T_GF2_16x)2, (BPU_T_GF2_16x) BPU_GF2_POLY_DEG_6);
149  }
150  else if (m == 11 && t == 50) {
151  rc = BPU_mathInitCtx(ctx, (BPU_T_GF2_16x)2, (BPU_T_GF2_16x) BPU_GF2_POLY_DEG_11);
152  }
153  else {
154  BPU_printError("Code params not supported. Supported only (m,t): (5,5), (6,6), (6,7), (11,50)");
155 
156  return BPU_EC_CODE_PARAMS_NOT_SUPPORTED;
157  }
158  return rc;
159 }

161 void BPU_codeFreeCtx(BPU_T_Code_Ctx **ctx) {
162  BPU_T_Code_Ctx *ctx_p = *ctx;
163 
164  if (!ctx_p) {
165  return;
166  }
167  switch (ctx_p->type) {
168  case BPU_EN_CODE_GOPPA:
169  BPU_goppaFreeSpec(ctx_p->code_spec->goppa);
170  free(ctx_p->code_spec->goppa);
171  break;
172  case BPU_EN_CODE_QCMDPC:
173  BPU_qcmdpcFreeSpec(ctx_p->code_spec->qcmdpc);
174  // free(ctx_p->code_spec->qcmdpc);
175  break;
176  default:
177  BPU_printError("Code type not supported: %d", ctx_p->type);
178  }
179  BPU_gf2VecFree(&ctx_p->e);
180  BPU_mathFreeCtx(&ctx_p->math_ctx);
181 
182  free(ctx_p->code_spec);
183  free(ctx_p);
184  *ctx = NULL;
185 }
