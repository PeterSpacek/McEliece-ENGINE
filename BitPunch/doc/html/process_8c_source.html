630 int BPU_decryptCCA2KobaraImaiA(BPU_T_Vector_GF2 *message, const BPU_T_Vector_GF2 *cipher, const BPU_T_McEliece_Ctx *ctx) {
631  uint16_t message_len;
632  int rc = 0;
633  BPU_T_Vector_GF2 z1, z1_, z2, z3, decoded, r, h, hash_r, hash_e, hh, temp;
634  int n, k;
635 
636  // calculate n, and k
637  n = ctx->pub_key.g_mat.n;
638  k = ctx->pub_key.g_mat.k;
639 
640 #ifdef BPU_DEBUG_DECRYPT
641  BPU_printDebug("###### DECRYPT ######");
642 #endif
643  // PRE MENSIE POLIA - deg 5 a menej - TENTO VYPOCET NESEDI (preto pub_key->m - 3)
644  message_len = ctx->max_pt_len_bit;
645 
646 #ifdef BPU_DEBUG_DECRYPT
647  BPU_printDebug("cipher text:");
648  BPU_printGf2Vec(cipher);
649 #endif
650  rc = BPU_gf2VecCropA(&z1, cipher, 0, (0x1 << (ctx->a_data.mod_deg)) -1);
651  rc = BPU_gf2VecCropA(&z1_, cipher, 0, (0x1 << (ctx->a_data.mod_deg)) -1);
652  rc = BPU_gf2VecCropA(&z2, cipher, (0x1 << (ctx->a_data.mod_deg)) -1, message_len);
653  rc = BPU_gf2VecCropA(&z3, cipher, (0x1 << (ctx->a_data.mod_deg)) -1 + message_len, message_len);
654 
655 #ifdef BPU_DEBUG_DECRYPT
656  BPU_printDebug("z1:");
657  BPU_printGf2Vec(&z1);
658  BPU_printDebug("z2:");
659  BPU_printGf2Vec(&z2);
660  BPU_printDebug("z3:");
661  BPU_printGf2Vec(&z3);
662 #endif
663  BPU_decodeA(&z1_, &decoded, ctx);
664 
665 #ifdef BPU_DEBUG_DECRYPT
666  BPU_printDebug("decoded error:");
667  BPU_printGf2Vec(&decoded);
668 #endif
669  rc = BPU_gf2VecXor(&z1, &decoded);
670  // m: m*G = z1 ===> z1 = m
671 
672  /*Split m ′ to ( r , h , m ′′ ).
673  k−l bits l bits n−k bits
674  Reconstruct plaintext candidate m = z 2 ⊕ hash (r).
675  Determine check value h ′ = hash (m ∥ hash (e) ⊕ z 3 ).
676  if h ′ ≡ h then
677  15:
678  return m
679  16: end if
680  17: return “Error”*/
681  rc = BPU_gf2VecCropA(&r, &z1, n, k - message_len);
682  rc = BPU_gf2VecCropA(&h, &z1, n + k - message_len, message_len);
683 
684  rc = BPU_gf2VecHashA(&hash_r, &r, message_len);
685  rc = BPU_gf2VecXor(&z2, &hash_r);
686 
687  // Determine check value h′ = hash (m || hash (e) ⊕ z3 )
688  rc = BPU_gf2VecHashA(&hash_e, &decoded, message_len);
689  rc = BPU_gf2VecXor(&hash_e, &z3);
690  rc = BPU_gf2VecConcatA(&temp, &z2, &hash_e);
691  rc = BPU_gf2VecHashA(&hh, &temp, message_len);
692 
693 #ifdef BPU_DEBUG_DECRYPT
694  BPU_printDebug("plaintext with padding:");
695  BPU_printGf2Vec(&z2);
696 #endif
697  if (BPU_gf2VecCmp(&h, &hh)) {
698  BPU_printError("check value not correct");
699 
700  rc = 1;
701  }
702  else {
703  // delete padding
704  rc = BPU_delPaddingA(message, &z2);
705 
706  #ifdef BPU_DEBUG_DECRYPT
707  BPU_printDebug("plaintext w/o padding:");
708  BPU_printGf2Vec(message);
709  #endif
710  }
711  BPU_freeVecGF2(&z1, 0);
712  BPU_freeVecGF2(&z2, 0);
713  BPU_freeVecGF2(&z3, 0);
714  BPU_freeVecGF2(&decoded, 0);
715  BPU_freeVecGF2(&r, 0);
716  BPU_freeVecGF2(&h, 0);
717  BPU_freeVecGF2(&hash_r, 0);
718  BPU_freeVecGF2(&hash_e, 0);
719  BPU_freeVecGF2(&hh, 0);
720  BPU_freeVecGF2(&temp, 0);
721 
722 #ifdef BPU_DEBUG_DECRYPT
723  BPU_printDebug("###### DECRYPT DBPU_ONE ######");
724 #endif
725  return rc;
726 }
